943  Commit.prototype.arrow = function Arrow () {
944  // Options
945  var size = this.template.arrow.size;
946  var color = this.template.arrow.color || this.branch.color;
947 
948  // Angles calculation
949  var alpha = Math.atan2(
950  this.parentCommit.y - this.y,
951  this.parentCommit.x - this.x
952  );
953 
954  // Merge & Fork case
955  if ( this.type === "mergeCommit" || this === this.branch.commits[ 0 ] /* First commit */ ) {
956  alpha = Math.atan2(
957  this.template.branch.spacingY * (this.parentCommit.branch.column - this.branch.column) + this.template.commit.spacingY * (this.showLabel ? 2 : 1),
958  this.template.branch.spacingX * (this.parentCommit.branch.column - this.branch.column) + this.template.commit.spacingX * (this.showLabel ? 2 : 1)
959  );
960  color = this.parentCommit.branch.color;
961  }
962 
963  var delta = Math.PI / 7; // Delta between left & right (radian)
964 
965  // Top
966  var h = this.template.commit.dot.size + this.template.arrow.offset;
967  var x1 = h * Math.cos( alpha ) + this.x;
968  var y1 = h * Math.sin( alpha ) + this.y;
969 
970  // Bottom left
971  var x2 = (h + size) * Math.cos( alpha - delta ) + this.x;
972  var y2 = (h + size) * Math.sin( alpha - delta ) + this.y;
973 
974  // Bottom center
975  var x3 = (h + size / 2) * Math.cos( alpha ) + this.x;
976  var y3 = (h + size / 2) * Math.sin( alpha ) + this.y;
977 
978  // Bottom right
979  var x4 = (h + size) * Math.cos( alpha + delta ) + this.x;
980  var y4 = (h + size) * Math.sin( alpha + delta ) + this.y;
981 
982  this.context.beginPath();
983  this.context.fillStyle = color;
984  this.context.moveTo( x1, y1 ); // Top
985  this.context.lineTo( x2, y2 ); // Bottom left
986  this.context.quadraticCurveTo( x3, y3, x4, y4 ); // Bottom center
987  this.context.lineTo( x4, y4 ); // Bottom right
988  this.context.fill();
989  };
